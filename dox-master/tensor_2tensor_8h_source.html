652  temp->data_ = temp->allocate(n);
653  try {
654  // need to construct elements of data_ using placement new in case its
655  // default ctor is not trivial N.B. for fundamental types and standard
656  // alloc this incurs no overhead (Eigen::aligned_alloc OK also)
657  auto* data_ptr = temp->data_;
658  for (ordinal_type i = 0; i != n; ++i, ++data_ptr)
659  new (static_cast<void*>(data_ptr)) value_type;
660 
661  ar& madness::archive::wrap(temp->data_, n);
662  ar & temp->range_;
663  } catch (...) {
664  temp->deallocate(temp->data_, n);
665  throw;
666  }

850  template <typename Perm,
851  typename = std::enable_if_t<detail::is_permutation_v<Perm>>>
852  Tensor_ permute(const Perm& perm) const {
853  constexpr bool is_tot = detail::is_tensor_of_tensor_v<Tensor_>;
854  [[maybe_unused]] constexpr bool is_bperm =
855  detail::is_bipartite_permutation_v<Perm>;
856  // tile ops pass bipartite permutations here even if this is a plain tensor
857  // static_assert(is_tot || (!is_tot && !is_bperm), "Permutation type does
858  // not match Tensor_");
859  if constexpr (!is_tot) {
860  if constexpr (is_bperm) {
861  TA_ASSERT(inner_size(perm) == 0); // ensure this is a plain permutation
862  return Tensor_(*this, outer(perm));
863  } else
864  return Tensor_(*this, perm);
865  } else {
866  // If we have a ToT we need to apply the permutation in two steps. The
867  // first step is identical to the non-ToT case (permute the outer modes)
868  // the second step does the inner modes
869  Tensor_ rv(*this, outer(perm));
870  if constexpr (is_bperm) {
871  if (inner_size(perm) != 0) {
872  auto inner_perm = inner(perm);
873  Permute<value_type, value_type> p;
874  for (auto& inner_t : rv) inner_t = p(inner_t, inner_perm);
875  }
876  }
877  return rv;
